# test train function 
trte<-function(dataf,p=.8){
  
  tr_id<-sample(1:nrow(dataf),p*nrow(dataf))
  
  train<-dataf[tr_id,]
  
  test<-dataf[-tr_id,]
  
  return(list(train,test))
}


trte_seed<-function(dataf,p=.8,seedv=123){
  
  set.seed(seedv)
  tr_id<-sample(1:nrow(dataf),p*nrow(dataf))
  
  train<-dataf[tr_id,]
  
  test<-dataf[-tr_id,]
  
  return(list(train,test))
}
# function to compute test set RMSE on the spfc variable given a model and test set
crmse<-function(model,test,var="Sweetpotato_Flavor"){
  
  preds<-predict(model,test)
  
  obs<-test[,var]
  
  rmse<-sqrt( mean((preds-obs)^2))
  
  return(rmse)
}
# gets the best tuned model from a caret fitted model
get_best_result = function(caret_fit) {
  best = which(rownames(caret_fit$results) == rownames(caret_fit$bestTune))
  best_result = caret_fit$results[best, ]
  rownames(best_result) = NULL
  best_result
}

Reading in Data

  • We only deal with the 40 observations that have record for at least one VOC

  • Testing and training sets allow us to evaluate the performance of statistical models.

  • A testing set provides an unbiased estimate of the models performance on new, unseen data.

  • A training set is used to fit the model parameters, and having separate training and testing sets helps to prevent overfitting, where a model learns to perform well on the training data but does not generalize well to new data.

  • Without a testing set, we might mistakenly believe that our model is performing well when in fact it is overfitting to the training data.

  • A testing set can also be used to compare the performance of different models or variations of the same model.

  • In addition, a validation set can be used during the training process to tune hyperparameters or select the best model based on performance on the validation set.

  • Ultimately, having separate training and testing sets helps us build more accurate and reliable statistical models.

Cross Validation is a process that is used to pick the optimal hyperparameters for a model.

PLS

  • PCR makes new columns to describe as much as possible the variation in the data of VOC concentrations.

  • However nothing guarantees that the principal components, which explain VOCs optimally, will be relevant for the prediction of Sweet Potato Flavor.

  • Solution: incorporate information from Sweet Potato Flavor when choosing the projection. We thus choose a projection that describes as well as possible the covariation of data VOCs and labels Sweet Potato Flavor.

Advantages of Partial Least Squares:

  1. It can handle high-dimensional data with multicollinearity.
  2. It can handle small sample sizes with more predictor variables than observations.
  3. It can perform dimensionality reduction while preserving most of the variation in the data.
  4. It can deal with both continuous and categorical predictor variables.
  5. It can handle missing data by using available data to estimate missing values.
  6. It can be used for both regression and classification problems.

Disadvantages of Partial Least Squares:

  1. It may not perform well when there is no relationship between the predictors and the response variable.
  2. It may produce unstable models when the sample size is too small or the number of predictors is too large.
  3. It may not be able to identify all the important predictors in the data.
  4. It may not be as interpretable as other methods such as linear regression.
  5. It may require a significant amount of computation time and resources for large datasets.

Lasso

  • LASSO (Least Absolute Shrinkage and Selection Operator) is a regression analysis technique used for feature selection and regularization. It aims to reduce the complexity of a regression model by shrinking the coefficients of some predictors to zero, effectively selecting only the most important predictors.

  • LASSO is particularly useful in high-dimensional data sets where the number of predictors is much larger than the number of observations. By reducing the number of predictors, LASSO can improve the accuracy of the model and prevent overfitting.

  • This model selects important variables! It does this by setting not important ones to zero

\[\texttt{minimize}: \sum_{i=1}^n (y_i - \beta_0 - \sum_{j=1}^p x_{ij}\beta_j)^2 + \lambda \sum_{j=1}^p |\beta_j|\]

In plain old Regression we try and minimize the below function to get our slopes. See how it is similar to the LASSO function

\[\sum_{i=1}^n (y_i - \beta_0 - \sum_{j=1}^p x_{ij}\beta_j)^2\]

bootconfint<-function(x,B=1000,level=.95){
  # bootstrap conf int 
  # data is a vector
  
  many_means=replicate(B,mean(sample(x,length(x),replace=T)))
  
  # quantile 
  upper<-quantile(many_means,level+(1-level)/2)
  lower<-quantile(many_means,(1-level)/2)
  
  return(c(lower,upper))
}

get_repped_ennet_importances<-function(dataf,nfit){
  
# list of important coefs from each nfit 
Important_coefs<-vector(mode = "list", length = nfit)

for(iter in 1:nfit){
  #train test data
  train_test<-trte_seed(dataf,seedv=iter)
  train<-train_test[[1]]
  test<-train_test[[2]]

  
  # elastic net is trained
    
   cv <- trainControl(method = "cv",number = 5)            
    tuneGridenet <- expand.grid(alpha = seq(0,1,length=10), lambda = seq(0.0001, 1, length = 40))
    
    
    enet_fit <- train(
      Sweetpotato_Flavor ~., 
      data = train,
      method = "glmnet",
      trControl =cv,
      tuneGrid =tuneGridenet,
      preProcess=c("scale")
    )
  
    
  # importance of variables from one lasso run 
      
    
  # best lambda
  lambda_use <- min(enet_fit$finalModel$lambda[enet_fit$finalModel$lambda >=                    enet_fit$bestTune$lambda])
  #position in df
  position <- which(enet_fit$finalModel$lambda == lambda_use)
  # get all coefs from this model 
  df_enet_coef<-data.frame(coef(enet_fit$finalModel)[, position])
  # select important rows that are postive! 
  colnames(df_enet_coef)<-c("Value")
  df_enet_coef%>%filter(Value!=0)
  # for the iteration of the list store "important" values
  Important_coefs[[iter]]<-rownames(df_enet_coef%>%filter(Value!=0))
}
  
  
  agg_important_coefs<-Reduce(intersect,Important_coefs)




return(agg_important_coefs)
}

get_repped_ennet_importances<-function(dataf,nfit){
  
# list of important coefs from each nfit 
Important_coefs<-vector(mode = "list", length = nfit)

for(iter in 1:nfit){
  #train test data
  train_test<-trte(dataf)
  train<-train_test[[1]]
  test<-train_test[[2]]

  
  # elastic net is trained
    
   cv <- trainControl(method = "cv",number = 5)            
    tuneGridenet <- expand.grid(alpha = seq(0,1,length=10), lambda = seq(0.0001, 1, length = 40))
    
    
    enet_fit <- train(
      Sweetpotato_Flavor ~., 
      data = train,
      method = "glmnet",
      trControl =cv,
      tuneGrid =tuneGridenet,
      preProcess=c("scale")
    )
  
    
  # importance of variables from one lasso run 
      
    
  # best lambda
  lambda_use <- min(enet_fit$finalModel$lambda[enet_fit$finalModel$lambda >=                    enet_fit$bestTune$lambda])
  #position in df
  position <- which(enet_fit$finalModel$lambda == lambda_use)
  # get all coefs from this model 
  df_enet_coef<-data.frame(coef(enet_fit$finalModel)[, position])
  # select important rows that are postive! 
  colnames(df_enet_coef)<-c("Value")
  df_enet_coef%>%filter(Value!=0)
  # for the iteration of the list store "important" values
  Important_coefs[[iter]]<-rownames(df_enet_coef%>%filter(Value!=0))
}
  
  
  agg_important_coefs<-Reduce(intersect,Important_coefs)




return(agg_important_coefs)
}

# test train function 
trte<-function(dataf,p=.8){
  
  tr_id<-sample(1:nrow(dataf),p*nrow(dataf))
  
  train<-dataf[tr_id,]
  
  test<-dataf[-tr_id,]
  
  return(list(train,test))
}

# this function splits data based on a set seed 
trte_seed<-function(dataf,p=.8,seedv=123){
  
  set.seed(seedv)
  tr_id<-sample(1:nrow(dataf),p*nrow(dataf))
  
  train<-dataf[tr_id,]
  
  test<-dataf[-tr_id,]
  
  return(list(train,test))
}

This function computes the root mean square error. We use this measure to compute error.

# function to compute test set RMSE on the spfc variable given a model and test set
crmse<-function(model,test,var="Sweetpotato_Flavor"){
  
  preds<-predict(model,test)
  
  obs<-test[,var]
  
  rmse<-sqrt( mean((preds-obs)^2))
  
  return(rmse)
}

This function will let us get the best result in terms of cross validation error.

# gets the best tuned model from a caret fitted model
get_best_result = function(caret_fit) {
  best = which(rownames(caret_fit$results) == rownames(caret_fit$bestTune))
  best_result = caret_fit$results[best, ]
  rownames(best_result) = NULL
  best_result
}
